CREATE OR REPLACE FUNCTION public.is_admin()
RETURNS BOOLEAN LANGUAGE sql STABLE AS $$
  SELECT COALESCE(auth.jwt() ->> 'email', '') = 'rashtriya.73@gmail.com'
$$;

-- Users: allow admin to update any row
DROP POLICY IF EXISTS "users_update" ON public.users;
CREATE POLICY "users_update" ON public.users
  FOR UPDATE USING (auth.uid() = id OR public.is_admin());

-- Redeems: admin sees all, can update status
DROP POLICY IF EXISTS "redeems_select" ON public.redeem_requests;
CREATE POLICY "redeems_select" ON public.redeem_requests
  FOR SELECT USING (auth.uid() = user_id OR public.is_admin());

DROP POLICY IF EXISTS "redeems_update" ON public.redeem_requests;
CREATE POLICY "redeems_update" ON public.redeem_requests
  FOR UPDATE USING (public.is_admin());

-- Chat: admin can delete messages
DROP POLICY IF EXISTS "chat_delete" ON public.world_chat;
CREATE POLICY "chat_delete" ON public.world_chat
  FOR DELETE USING (public.is_admin());
